#### 太简单了 

```python
class Solution:
    def romanToInt(self, s: str) -> int:
        result = 0
        length = len(s)
        Functions = {'I': 1, 'V': 5, 'X': 10,'L': 50, 'C': 100, 'D': 500,'M': 1000}
        
        for i in range(length):
            if i + 1 < length and Functions[ s[i] ] >= Functions[ s[i + 1] ]:
                result += Functions[ s[i] ]
                
            elif i + 1 < length and Functions[ s[i] ] < Functions[ s[i + 1] ]:
                result -=  Functions[ s[i] ]
                
            elif i + 1 >= length:
                result += Functions[ s[i] ]
                
        return result
```
