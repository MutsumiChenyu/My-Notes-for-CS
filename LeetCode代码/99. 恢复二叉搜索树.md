```python
class Solution:
    def recoverTree(self, root: Optional[TreeNode]) -> None:
       
        def dfs(lst, root):
            if not root:
                return
            dfs(lst, root.left)
            lst.append(root)
            dfs(lst, root.right)
            return

        lst = list()
        dfs(lst, root)
        n = len(lst)
        Change1 = 0
        Change2 = 0
  
        for i in range(1, n):
            if lst[i].val <= lst[i - 1].val:
                if Change1 == 0 and Change2 == 0:
                    Change1 = i - 1
                    Change2 = i
                elif Change2 != 0:
                    Change2 = i
                    break

        tmp = lst[Change1].val
        lst[Change1].val = lst[Change2].val
        lst[Change2].val = tmp
```
