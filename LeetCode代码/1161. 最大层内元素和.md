```python
from collections import deque
class Solution:
    def maxLevelSum(self, root: Optional[TreeNode]) -> int:

        if not root:
            return 0

        quene = deque([root])
        x = 1
        GeShu = 1
        MaxResult = root.val
        MaxRow = 1

        while quene:
            result = 0
            temp = GeShu
            GeShu = 0
  
            for i in range(temp):
                node = quene.popleft()
                result += node.val

                if node.left:
                    GeShu += 1
                    quene.append(node.left)
                if node.right:
                    GeShu += 1
                    quene.append(node.right)

            if result > MaxResult:
                MaxRow = x
                MaxResult = result
                
            x += 1

        return MaxRow
```
