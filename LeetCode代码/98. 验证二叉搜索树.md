class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        A = list()
        
        def dfs(List, root):
            if not root:
                return
            dfs(List, root.left)
            List.append(root.val)
            dfs(List, root.right)
            return

        dfs(A, root)
        n = len(A)

        for i in range(1, n):
            if A[i] <= A[i - 1]:
                return False
                
        return True
